# S1_04_字符设备驱动

## 一、cdev 生命周期

```c
static dev_t devno;
static struct cdev skel_cdev;

// init
alloc_chrdev_region(&devno, 0, 1, "skel");
cdev_init(&skel_cdev, &skel_fops);
cdev_add(&skel_cdev, devno, 1);

// exit
cdev_del(&skel_cdev);
unregister_chrdev_region(devno, 1);
```

**动态分配 vs 静态**：`alloc_chrdev_region` 动态分配，避免主设备号冲突，不要用 `MKDEV` 静态申请。

## 二、file_operations 关键回调

```c
static struct file_operations skel_fops = {
    .owner = THIS_MODULE,
    .open = skel_open,
    .read = skel_read,
    .write = skel_write,
    .unlocked_ioctl = skel_ioctl,
    .mmap = skel_mmap,
    .release = skel_release,
};
```

**read 返回值**：返回实际复制的字节数，不是请求的字节数。

**read 实现模板**：
```c
static ssize_t skel_read(struct file *filp, char __user *buf,
                         size_t count, loff_t *ppos)
{
    if (*ppos >= buf_len) return 0;
    if (*ppos + count > buf_len) count = buf_len - *ppos;
    if (copy_to_user(buf, buffer + *ppos, count))
        return -EFAULT;
    *ppos += count;
    return count;
}
```

**write 实现模板**：
```c
static ssize_t skel_write(struct file *filp, const char __user *buf,
                          size_t count, loff_t *ppos)
{
    if (count > BUF_SIZE - 1) count = BUF_SIZE - 1;
    if (copy_from_user(buffer, buf, count))
        return -EFAULT;
    buffer[count] = '\0';
    buf_len = count;
    return count;
}
```

## 三、copy_from_user 必须检查返回值

```c
// 正确：检查返回值
if (copy_from_user(buffer, buf, count))
    return -EFAULT;  // 用户地址无效

// 错误：忽略返回值
copy_from_user(buffer, buf, count);  // ❌ 可能在无效地址上崩溃
```

**copy_to_user 同样必须检查**。

## 四、ioctl 设计

```c
static long skel_ioctl(struct file *filp, unsigned int cmd, unsigned long arg)
{
    int val;
    switch (cmd) {
    case SKEL_GET_COUNT:       return buf_len;
    case SKEL_SET_MODE:
        if (copy_from_user(&val, (int __user *)arg, sizeof(val)))
            return -EFAULT;
        mode = val;
        return 0;
    case SKEL_RESET:
        buf_len = 0;
        memset(buffer, 0, BUF_SIZE);
        return 0;
    default:
        return -EINVAL;
    }
}
```

**ioctl 编号**（避免和已有命令冲突）：
```c
#define SKEL_GET_COUNT   _IOR('S', 0x00, int)
#define SKEL_SET_MODE    _IOW('S', 0x01, int)
#define SKEL_RESET       _IO('S',  0x02)
```

## 五、mmap（PCIe 驱动特别有用）

把 BAR 寄存器直接映射到用户态，用户程序直接读写：

```c
static int skel_mmap(struct file *filp, struct vm_area_struct *vma)
{
    unsigned long pfn = virt_to_phys(buffer) >> PAGE_SHIFT;
    if (vm_insert_pfn(vma, vma->vm_start, pfn))
        return -EAGAIN;
    return 0;
}
```

用户态：`mmap(NULL, size, PROT_READ|PROT_WRITE, MAP_SHARED, fd, 0)` 之后直接读写内存，即读写 BAR 寄存器。

## 六、工程实践：PCIe 驱动的典型用户态 API

```bash
open("/dev/pcie_ep0")              # 打开设备
ioctl(fd, PCIE_GET_BAR_INFO)         # 获取BAR信息
mmap(fd, BAR0, SIZE)                # 把BAR映射到用户空间，直接读写寄存器
read(fd, buf, 4)                    # 读寄存器
write(fd, buf, 4)                   # 写寄存器
ioctl(fd, PCIE_TRIGGER_MSI)          # 触发MSI
close(fd)
```
